13 // TMC429 Motor numbers to be combined with the addresses of the motor register set
14 // (e.g. TMC429_IDX_XTARGET|TMC429_MOTOR1).
15 #define TMC429_MOTOR0 0
16 #define TMC429_MOTOR1 0x20
17 #define TMC429_MOTOR2 0x40
18 #define TMC429_MOTOR(motor) (motor<<5)
19
20 // TMC429 Motor Register Set
21 #define TMC429_IDX_XTARGET(motor) (0x00|TMC429_MOTOR(motor))
22 #define TMC429_IDX_XACTUAL(motor) (0x02|TMC429_MOTOR(motor))
23 #define TMC429_IDX_VMIN(motor) (0x04|TMC429_MOTOR(motor))
24 #define TMC429_IDX_VMAX(motor) (0x06|TMC429_MOTOR(motor))

26 #define TMC429_IDX_VACTUAL(motor) (0x0A|TMC429_MOTOR(motor))
27 #define TMC429_IDX_AMAX(motor) (0x0C|TMC429_MOTOR(motor))
28 #define TMC429_IDX_AACTUAL(motor) (0x0E|TMC429_MOTOR(motor))
