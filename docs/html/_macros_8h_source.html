12/* Cast a n bit signed int to a 32 bit signed int
13 * This is done by checking the MSB of the signed int (Bit n).
14 * If it is 1, the value is negative and the Bits 32 to n+1 are set to 1
15 * If it is 0, the value remains unchanged
16 */
17#define CAST_Sn_TO_S32(value, n) ((value) | (((value) & ((uint32_t)1<<((n)-1)))? ~(((uint32_t)1<<(n))-1) : 0 ))
