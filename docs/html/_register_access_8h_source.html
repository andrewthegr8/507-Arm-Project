8/*
9 * The permission system aims to allow a general-purpose implementation for
10 * all common hardware register usages. This includes:
11 * - Trivial Cases: Read, Write, Read & Write
12 *
13 * - Read & Write accesses that route to different values/functions of a chip.
14 * (e.g. serial communication, where read/write corresponds to RX/TX)
15 * - Read to clear, write to clear. This does not directly affect the access,
16 * but can be used to implement a software shadow register for flags
17 * (ORing the read value into a shadow register instead of overwriting).
18 * - Registers with default values that are not known (e.g. Factory configuration
19 * values that should not be overwritten by default).
20 */
21
22#ifndef TMC_HELPERS_REGISTERACCESS_H
23#define TMC_HELPERS_REGISTERACCESS_H
24
25// Register access bits
26/* Lower nibble is used for read/write, higher nibble is used for
27 * special case registers. This makes it easy to identify the read/write
28 * part of the permissions in a hexadecimal permission number.
29 * The dirty bit will only ever be set at runtime, so we keep the easily
30 * readable lower nibble.
31 */
32#define TMC_ACCESS_NONE 0x00
33
34#define TMC_ACCESS_READ 0x01
35#define TMC_ACCESS_WRITE 0x02
36 // 0x04 is currently unused
37#define TMC_ACCESS_DIRTY 0x08 // Register has been written since reset -> shadow register is valid for restore
38
39// Special Register bits
40#define TMC_ACCESS_RW_SPECIAL 0x10 // Read and write are independent - different values and/or different functions
41#define TMC_ACCESS_FLAGS 0x20 // Register has read or write to clear flags.
42#define TMC_ACCESS_HW_PRESET 0x40 // Register has hardware presets (e.g. Factory calibrations) - do not write a default value
43 // 0x80 is currently unused
44
45// Permission combinations
46#define TMC_ACCESS_RW (TMC_ACCESS_READ | TMC_ACCESS_WRITE) // 0x03 - Read and write
47#define TMC_ACCESS_RW_SEPARATE (TMC_ACCESS_RW | TMC_ACCESS_RW_SPECIAL) // 0x13 - Read and write, with separate values/functions
48#define TMC_ACCESS_R_FLAGS (TMC_ACCESS_READ | TMC_ACCESS_FLAGS) // 0x21 - Read, has flags (read to clear)
49#define TMC_ACCESS_RW_FLAGS (TMC_ACCESS_RW | TMC_ACCESS_FLAGS) // 0x23 - Read and write, has flags (read or write to clear)
50#define TMC_ACCESS_W_PRESET (TMC_ACCESS_WRITE | TMC_ACCESS_HW_PRESET) // 0x42 - Write, has hardware preset - skipped in reset routine
51#define TMC_ACCESS_RW_PRESET (TMC_ACCESS_RW | TMC_ACCESS_HW_PRESET) // 0x43 - Read and write, has hardware presets - skipped in reset routine
52
53// Helper macros
54#define TMC_IS_READABLE(x) ((x) & TMC_ACCESS_READ)
55#define TMC_IS_WRITABLE(x) ((x) & TMC_ACCESS_WRITE)
56#define TMC_IS_DIRTY(x) ((x) & TMC_ACCESS_DIRTY)
57#define TMC_IS_PRESET(x) ((x) & TMC_ACCESS_HW_PRESET)
58#define TMC_IS_RESETTABLE(x) (((x) & (TMC_ACCESS_W_PRESET)) == TMC_ACCESS_WRITE) // Write bit set, Hardware preset bit not set
59#define TMC_IS_RESTORABLE(x) (((x) & TMC_ACCESS_WRITE) && (!(x & TMC_ACCESS_HW_PRESET) || (x & TMC_ACCESS_DIRTY))) // Write bit set, if it's a hardware preset register, it needs to be dirty
60
61// Struct for listing registers that have constant contents which we cannot
62// obtain by reading them due to the register not being read-back.
63typedef struct
64{
65 uint8_t address;
66 uint32_t value;
67} TMCRegisterConstant;
68
69// Helper define:
70// Most register permission arrays are initialized with 128 values.
71// In those fields its quite hard to have an easy overview of available
72// registers. For that, ____ is defined to 0, since 4 underscores are
73// very easy to distinguish from the 2-digit hexadecimal values.
74// This way, the used registers (permission != ACCESS_NONE) are easily spotted
75// amongst unused (permission == ACCESS_NONE) registers.
76#define ____ 0x00
77
78// Helper define:
79// Default reset values are not used if the corresponding register has a
80// hardware preset. Since this is not directly visible in the default
81// register reset values array, N_A is used as an indicator for a preset
82// value, where any value will be ignored anyways (N_A: not available).
83#define N_A 0

85#endif /* TMC_HELPERS_REGISTERACCESS_H */
